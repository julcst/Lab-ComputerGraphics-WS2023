    \note[item]{If we take a look at the \textbf{paths} that the light for a specific incident direction can take, we can see that because of \textbf{multiple reflections and transmissions inside} the stack of layers the light transport can be \textbf{very intricate}.}

    \note[item]{Therefore, modelling the light transport in layered materials is \textbf{complex problem}.}

\begin{frame}
    \frametitle{Microfacet model}
    \begin{equation*}
        \rho(\omega_i, \omega_o) = \frac{F(\omega_i,h) G(\omega_i,\omega_o) D(h)}{4 \dotproduct{\omega_i}{n} \dotproduct{\omega_o}{n}}
    \end{equation*}
    \begin{itemize}
        \item<1-> $\rho$ microfacet BRDF
        \item<2-> $F$ Fresnel term
        \item<2-> $G$ shadowing-masking function
        \item<2-> $D$ normal distribution function
        \item<3-> $\omega_i$ incident direction / $\omega_o$ outgoing direction
        \item<3-> $h$ half-vector, $n$ normal
        \item<3-> $\dotproduct{}{}$ dot product
    \end{itemize}
    %%%%%%%%%%%%%%%%%
    \note[item]{In physically based rendering, typically a \textbf{microfacet model} is used. Consisting of a \textbf{diffuse and a specular component} this is able to reproduce many appealing surfaces, but this is \textbf{not enough} to model layered materials.}
    \note[item]{\click On the slide you can see how the \textbf{BRDF of a microfacet model} is defined by the \textbf{Fresnel} term F, the \textbf{shadow-masking} function G and the \textbf{normal distribution} function D, for example GGX.}
    \note[item]{\click For \textbf{notation} we will use omega i for the incident direction and omega o for the outgoing direction, with h for the half-vector.}
\end{frame}

\begin{frame}
    \frametitle{Goals}
    \begin{itemize}
        \item Physically based
        \item Arbitrary number of layers
        \item Multiple scattering
        \item Energy conserving
        \item Texturing
        \item Importance Sampling
        \item Little storage
        \item Short precomputation
        \item Efficient evaluation
    \end{itemize}
    %%%%%%%%%%%%%%%%%
    \note[item]{These are the \textbf{desirable goals} for a model for layered materials.}
    \note[item]{It should be \textbf{physically based}, allow an \textbf{arbitrary number of layers}, account for \textbf{multiple scattering}. Of course, it also should be \textbf{energy conserving}, support \textbf{texturing} and \textbf{importance sampling}. Additionally, as little as possible \textbf{storage} and \textbf{precomputation} should be required and at best it should be \textbf{efficient to evaluate} so that it can be used for real-time applications.}
    \note[item]{There have been \textbf{several previous works} to the one we present today by Belcour. However, these only \textbf{achieved a sub set} of these goals.}
\end{frame}

\subsection{Overview}

%---------------------------------------------------------
\begin{frame}
    \frametitle{Statistics of a GGX lobe}
    \begin{itemize}
        \item<1-> bijective mapping between GGX parameters and directional statistics
        \item<2-> {
            \begin{minipage}{\linewidth}
                \centering
                \includegraphics[width = 0.4\textwidth, keepaspectratio]{fromBelcour/projectivespace.png}
                \captionof{figure}{Projective space. Image source: \textcite{belcour2018}}
            \end{minipage}
        }
    \end{itemize}
    %%%%%%%%%%%%%%%%%
    \note[item]{Belcour uses a \textbf{statistical framework for approximating the light transport} in layered materials.}
    \note[item]{The main insight is that a \textbf{bijective mapping} between the GGX parameters and the directional statistics of a GGX lobe exists. \click}
    \note[item]{This can be seen when we take a look at the \textbf{illustration} on the slide. If we \textbf{study the GGX lobe} for a specific incident direction omega i, we can see, that if the lobe is \textbf{projected} on to the base plane, it has a circular shape. Therefore, we can use a \textbf{circular distribution} and the statistics of this distribution in order to approximate the associated GGX lobe.}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Method overview}
    \begin{figure}
        \centering
        \includegraphics[width = \textwidth, keepaspectratio]{fromBelcour/addingdoubling.png}
        \caption{Method overview. Image source: \textcite{belcour2018}}
        \label{fig:overview}
    \end{figure}
    %%%%%%%%%%%%%%%%%
    \note[item]{Using this insight the \textbf{light transport inside a stack} of layers can be \textbf{described by a set of statistics}.}
    \note[item]{First we need so-called \textbf{atomic operators} to describe how the statics are changed by a light transport \textbf{interaction} with a rough interface, for example by a reflection.}
    \note[item]{Using the \textbf{adding-doubling method}, the combined reflected and transmitted statistics of a stack of \textbf{two interfaces} can be evaluated.}
    \note[item]{\textbf{Iteratively} applying this method on the whole stack of layers a vector of statistics can be computed which then can be used to \textbf{instantiate multiple GGX lobes} which approximate the appearance of the material.}
\end{frame}

\subsubsection{Rough reflection}

%---------------------------------------------------------
\begin{frame}
    \frametitle{Reflection by a rough surface}
    \framesubtitle{Energy}
    \textbf{Energy:}\\\bigskip
    Rendering equation:
    \begin{equation*}
        L_o(\omega_o) = \int_{\Omega} \rho_{GGX}(\omega_i, \omega_o) L_i(\omega_i) \dotproduct{\omega_i}{n} \,d\omega_i 
    \end{equation*}
    \pause
    Decoupling:
    \begin{equation*}
        \int_\Omega L_o(\omega_o) \,d\omega_o \simeq \FGD \times \int_\Omega L_i(\omega_i) \,d\omega_i
    \end{equation*}
    \pause
    Albedo:
    \begin{equation*}
        \FGD = \int_\Omega \rho_{GGX}(\omega_i, \omega_o) \dotproduct{\omega_i}{n} \,d\omega_o
    \end{equation*}
    %%%%%%%%%%%%%%%%%
    \note[item]{Now we will describe the \textbf{first operator} for the reflection by rough interface. We begin by looking at the \textbf{effect on the energy}.}
    \note[item]{Starting with the \textbf{rendering equation} we want to \textbf{decouple} the integral over the incoming light from the rest. So the \textbf{outgoing radiance} can be approximated like this. \click}
    \note[item]{The remaining \textbf{FGD} term describes the \textbf{albedo} of the surface. It is the factor that we use to change the incoming energy.}
    \note[item]{FGD can be \textbf{precomputed and stored}. While precomputation a scaling factor to account for multiple scattering can be incorporated in order to get an \textbf{energy conserving} model.}
    \note[item]{If precomputation is unwanted, it can also be \textbf{approximated} only using the Fresnel term.}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Reflection by a rough surface}
    \framesubtitle{Mean \& Variance}
    \textbf{Mean:} shift towards normal for high roughness, but negligible\\
    \bigskip
    \textbf{Variance:} increases but not linear w.r.t. roughness $\Rightarrow$ linear variance space
    %%%%%%%%%%%%%%%%%
    \note[item]{For higher roughness values the mean gets \textbf{shifted towards the normal}, but for the approximation this impact is negligible.}
    \note[item]{As said earlier the \textbf{variance} increases, however not with respect to GGX roughness, that's why we use the mapping of the roughness to use a linear transformation.}
\end{frame}
%---------------------------------------------------------

\begin{frame}
    \frametitle{Reflection by a rough surface}
    \framesubtitle{Summary}
    \centering
    \begin{tabular}{c c c c c}
        $\begin{aligned}
            e_r &= e_i \times \FGD\\
            \mu_r &= \reflect(\mu_i)\\
            \sigma_r &= \sigma_i + f(\alpha)
        \end{aligned}$
        & & & & 
        \raisebox{-25mm}{\includesvg[width = 0.3\textwidth]{operator_reflection.svg}}
    \end{tabular}
    %%%%%%%%%%%%%%%%%
    \note[item]{To \textbf{summarize} the first operator on how the statistics get affected by reflection, is described by these terms. For example the incident energy is multiplied by FGD.}
\end{frame}

                \end{overprint}
            \end{center}
        \end{column}
    \end{columns}
    %%%%%%%%%%%%%%%%%
    \note[item]{The \textbf{next operator} is refraction. Because of \textbf{energy conservation} the transmitted energy needs to be 1 - FGD. \click}
    \note[item]{For the mean the following can be observed. Looking at the \textbf{illustration} on the right, you can see that the \textbf{refracted incident direction} does not \textbf{align} with the \textbf{mean of the transmitted lobe}. However, just using the refracted incident direction is a good \textbf{approximation}.}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Refraction by a rough interface}
    \framesubtitle{Variance}
    \textbf{Variance:}
    \begin{figure}
        \includegraphics[width = 0.35\textwidth, keepaspectratio]{fromBelcour/fake_transmission.png}

    \end{figure}
    \pause
    \begin{equation*}
        s = \frac{1}{2} \left( \frac{\eta_{12} \dotproduct{\omega_i}{n} + \dotproduct{\omega_t}{n}}{\dotproduct{\omega_t}{n}} \right)
    \end{equation*}
    %%%%%%%%%%%%%%%%%
    \note[item]{Looking at the effect on the \textbf{variance}. It can be seen, that the refracted lobe can be approximated using a \textbf{scaled reflection from underneath} the interface. Therfore, the \textbf{black} incident direction, will be flipped underneath the interface and \textbf{scaled by the index of refraction} resulting in the \textbf{green} direction which  will be used for the reflection. For this \textbf{reflection a scaled roughness} will be used to get the blue distribution.\click}
    \note[item]{Belcour empirically found the following scaling factor for the roughness.}
\end{frame}

    \framesubtitle{Summary}

\end{frame}

\subsubsection{Absorption \& Scattering}

%---------------------------------------------------------
\begin{frame}
    \frametitle{Absorption}
    \framesubtitle{Energy, mean \& variance}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \onslide<1->{
                $\sigma_t = \sigma_a + \sigma_s$\\\bigskip
                \textbf{Energy:} Beer-Lambert's law: 
                \begin{equation*}
                    L_o(\omega_o) = \exp{\left(-\sigma_t \frac{h}{\dotproduct{\omega_o}{n}}\right)} L_i(\omega_o)
                \end{equation*} 
                \\\bigskip
            }
            \onslide<2->{
                \textbf{Mean \& variance:} Impact of absorption negligible
            }
        \end{column}
        \begin{column}{0.4\textwidth}
            \centering\includesvg[width = 0.7\textwidth]{operator_absorption.svg}
        \end{column}
    \end{columns}
    %%%%%%%%%%%%%%%%%
    \note[item]{Now we will take a look at the \textbf{effects of absorption and scattering} in participating media on the statistics. The effects are \textbf{parametrized} by the absorption cross-section sigma a and scattering cross-section sigma s, which result in the \textbf{transmittance} sigma t. We will treat absorption and scattering \textbf{separately} and begin with absorption.}
    \note[item]{The \textbf{attenuation} of the radiance is described by \textbf{Beer-Lamerts law}.}
    \note[item]{h is the \textbf{depth} of the layer. This fraction is the \textbf{length of the path} that the light takes through the medium.\click}
    \note[item]{There is some impact of absorption on the \textbf{mean and variance}, but this is negligible for the approximation.}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Scattering}
    \framesubtitle{Energy}
    \begin{columns}
        \begin{column}{0.7\textwidth}
            \onslide<1->{
                \textbf{Energy:} amount of radiance scattered accounting for in- and out-scattering: 
                \begin{equation*}
                    \begin{aligned}
                        L_o(\omega) &= \sigma_s \int_{0}^{h} \exp{\left(-\frac{\sigma_t t}{\dotproduct{\omega}{n}}\right)} L_s(t,\omega) \,dt \\
                        \text{with}\\
                        L_s(t,\omega) &= \int_{\mathcal{S}} p(\omega_i, \omega) \exp{\left(-\frac{\sigma_t (h-t)}{\dotproduct{\omega_i}{n}}\right)} L_i(\omega_i) \,d\omega_i
                    \end{aligned}
                \end{equation*}
                \\\bigskip
            }
            \onslide<2->{
                Approximation:
                \begin{equation*}
                    \int_{\Omega} L_o(\omega) \,d\omega \approx \sigma_s h \exp{\left(-\frac{\sigma_t h}{\dotproduct{\omega_o}{n}}\right)} \int_{\Omega} L_i(\omega) \,d\omega
                \end{equation*}
            }
        \end{column}
        \begin{column}{0.3\textwidth}
            \centering\includesvg[width = 0.7\textwidth]{operator_absorption.svg}
        \end{column}
    \end{columns}
    %%%%%%%%%%%%%%%%%
    \note[item]{For scattering, the \textbf{amount of radiance scattered accounting for in and out scattering} is described by these equations. We \textbf{integrate over all points} along the path through the medium and for each of these points we \textbf{integrate the incoming radiance} weighted by the \textbf{phase function p}. \click}
    \note[item]{If we now only account for \textbf{forward scattering} and also only evaluate it for the \textbf{mean direction of the incoming light}, those two \textbf{exponential functions cancel} out mostly as well as \textbf{p}, so there is no more the \textbf{dependence on t} and the integral over \textbf{h collapses} to simply the factor h. So the effect on the energy can be \textbf{approximated} using this term.}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Scattering}
    \framesubtitle{Mean \& variance}
    \onslide<1->{\textbf{Mean:} Approximation only accounts forward scattering $\Rightarrow$ impact on mean negligible\\\bigskip}
    \onslide<2->{\textbf{Variance:} increases w.r.t. width of phase function\\\bigskip}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \onslide<3->{
            Mapping from Henyey-Greenstein phase function anisotropy factor $g$ to variance \parencite{belcour2018}:
            \begin{equation*}
                \sigma_g = \left(\frac{1 - g}{g}\right)^{0.8} \frac{1}{1+g}
            \end{equation*}
            }
        \end{column}
        \begin{column}{0.4\textwidth}
            \onslide<2->\centering\includesvg[width = 0.9\textwidth]{phase_function.svg}
        \end{column}
    \end{columns}
    %%%%%%%%%%%%%%%%%
    \note[item]{For the \textbf{mean}, the impact is negligible as we only account for forward scattering. \click}
    \note[item]{The variance increases with respect to the \textbf{width of the phase function}. On the right, you can see an illustration of the \textbf{Henyey-Greenstein} phase function which is manly forward scattering. \click}
    \note[item]{Belcour empirically found this \textbf{mapping} from the anisotropy factor g of the Henyey-Greenstein phase function to variance.}
\end{frame}

\begin{frame}
    \frametitle{Absorption \& Scattering}
    \framesubtitle{Summary}
    \begin{equation*}
        \begin{aligned}
            e_t &= e_i \times \left[ \left(1 +  \frac{\sigma_s h}{\dotproduct{\omega_o}{n}}\right) \times \exp{\left(-\frac{\sigma_t h}{\dotproduct{\omega_o}{n}}\right)} \right]\\
            \mu_t &= \mu_i\\
            \sigma_t &= \sigma_i + \sigma_g \times \frac{\sigma_s}{\sigma_t}
        \end{aligned}
    \end{equation*}
    %%%%%%%%%%%%%%%%%
    \note[item]{To summarize, these terms describe the \textbf{combined effects} of absorption and scattering.}
\end{frame}

\begin{frame}
    \frametitle{Adding-doubling method}
    \begin{figure}
        \begin{overprint}
            \onslide<1>\centering\includesvg[width = 0.7\textwidth, keepaspectratio]{adding_doubling_0.svg}
            \onslide<2>\centering\includesvg[width = 0.7\textwidth, keepaspectratio]{adding_doubling_1.svg}
            \onslide<3>\centering\includesvg[width = 0.7\textwidth, keepaspectratio]{adding_doubling_2.svg}
            \onslide<4>\centering\includesvg[width = 0.7\textwidth, keepaspectratio]{adding_doubling_3.svg}
            \onslide<5>\centering\includesvg[width = 0.7\textwidth, keepaspectratio]{adding_doubling_4.svg}
            \onslide<6->\centering\includesvg[width = 0.7\textwidth, keepaspectratio]{adding_doubling_5.svg}
        \end{overprint}
        \label{fig:addingdoubling}
    \end{figure}
    \onslide<7->{\centering Evaluated for energy and variance. Impact on mean negligible.}
    %%%%%%%%%%%%%%%%%
    \note[item]{To do this, Belcour introduced the \textbf{adding doubling method} to account for multiple scattering. How this works I will describe now. If we have this stack of layers with a given incident light direction, we start by only looking at the \textbf{fist two interfaces}.\click}
    \note[item]{As seen earlier the \textbf{paths} light can take between them is complex.\click}
    \note[item]{So we use the \textbf{operators} defined in the previous section to summarize the paths using their reflected and transmitted \textbf{statistics}.\click}
    \note[item]{Now the first two interfaces will be treated as a \textbf{combined virtual interface} above the next interface. \click}
    \note[item]{The \textbf{transmitted statisitcs} of the first two interfaces will be used as the \textbf{incident} for the next iteration. And again the \textbf{operators} will be used to describe the light interactions.\click}
    \note[item]{\textbf{Iteratively} applied on all layers, always \textbf{reusing} the already combined statistics, we get a \textbf{vector} of reflected and transmitted statistics which summarizes the light transport inside the whole stack of layers.\click}
    \note[item]{The adding doubling method will be \textbf{separately} applied on energy and variance and as the impact on the \textbf{mean is negligible} as seen in the previous section, it will be ignored.}
\end{frame}

\subsubsection{Adding-doubling for energy}

%---------------------------------------------------------
\begin{frame}
    \frametitle{Radiance coefficients}
    Reflection and transmission coefficients:
    \begin{figure}
        \centering
        \includegraphics[width = 0.7\textwidth, keepaspectratio]{fromBelcour/coefficients.png}
        \caption{Reflection and transmission coefficients. Image source: \textcite{belcour2018}}
    \end{figure}
    %%%%%%%%%%%%%%%%%
    \note[item]{Starting with adding doubling \textbf{for energy}, we need the reflection and transmission \textbf{coefficients} of an interface. For the \textbf{notation}: r stand for reflection and t for transmission. The \textbf{order} of the numbers in the index denotes the \textbf{direction} of the light. For example r12 is the coefficient for a reflection of light coming from medium 1 towards medium 2.}
    \note[item]{On the \textbf{right} you can see the coefficients for a stack of two interfaces, which stand for the \textbf{sum of all interactions} between the two interfaces.}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Multiple scattering between two interfaces}
    \begin{columns}
        \begin{column}{0.7\textwidth}
            \onslide<1->{
                Account for all $TR^+T$ paths between two interfaces:
                \begin{equation*}
                    r_{13} = r_{12} + \sum_{k=0}^{\infty} t_{12} r_{23}^{k+1} r_{21}^{k} t_{21} 
                \end{equation*}
                \\\bigskip
            }
            \onslide<2->{
                Closed form of this geometric series:
                \begin{equation*}
                    r_{13} = r_{12} + \frac{t_{12} r_{23} t_{21}}{1 - r_{23} r_{21}}
                \end{equation*}
                \\\bigskip
            }
            \onslide<3->{
                Similar forms for $r_{31}$, $t_{13}$ and $t_{31}$
            }
        \end{column}
        \begin{column}{0.3\textwidth}
            \centering\includesvg[width = 0.9\textwidth]{cumulative_reflection_paths.svg}
        \end{column}
    \end{columns}
    %%%%%%%%%%%%%%%%%
    \note[item]{The coefficient r13 accounting for \textbf{multiple scattering} inside two interfaces can be \textbf{calculated} using this term.}
    \note[item]{The good thing about this is, that the \textbf{geometric series} has a closed form and therefore this equation can be simplified to this \click}
    \note[item]{This now can be \textbf{evaluated efficiently}. \click}
    \note[item]{\textbf{Similar forms} can be derived for the other direction and transmission.}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Adding-doubling for energy}
    \begin{itemize}
        \item<1-> iteratively evaluate $r_{13}$, $r_{31}$, $t_{13}$ and $t_{31}$ \bigskip
        \item<2->{ 
            use statistical operators as coefficients, e.g. for a rough interface:
            \begin{equation*}
                \begin{aligned}
                    r_{23} &= r_{32} = \FGD\\
                    t_{23} &= t_{32} = 1-\FGD
                \end{aligned}
            \end{equation*}
            \bigskip
        }
        \item<3-> doubling: virtual layer twice the depth
    \end{itemize}
    %%%%%%%%%%%%%%%%%
    \note[item]{\textbf{Iteratively} evaluating the coefficients for all interfaces of the stack of layers and always \textbf{reusing} the previously calculated coefficients of the combined interfaces above, we get the statistics for the energy for the whole stack.\click}
    \note[item]{For the \textbf{coefficients} of the \textbf{currently} evaluated interface we use the \textbf{operators} defined in the previous section. So for example FGD as the reflection coefficient.\click}
    \note[item]{The \textbf{doubling} step can be used to duplicate a layer of a small depth to create a layer of an \textbf{arbitrary depth}.}
\end{frame}
%---------------------------------------------------------

\subsubsection{Adding-doubling for variance}

%---------------------------------------------------------
\begin{frame}
    \frametitle{Adding-doubling for variance}
    \begin{itemize}
        \item similar to adding-doubling for energy, using average energy\bigskip
        \item scaling by energy\bigskip
        \item build on property: the variance of a weighted sum is the weighted sum of the variances
    \end{itemize}
    %%%%%%%%%%%%%%%%%
    \note[item]{The adding-doubling method \textbf{for variance} is similar to the one for energy. However, the variance needs to be \textbf{scaled by the energy}. For this the \textbf{average} energy is used.}
    \note[item]{It is build upon the \textbf{property} that the variance of a weighted sum is the weighed sum of the individual variances.}
\end{frame}

\subsection{BRDF}

%---------------------------------------------------------
\begin{frame}
    \frametitle{BRDF}
    \begin{itemize}
        \setlength\itemsep{1em}
        \item<1-> evaluate adding-doubling method with $e_i = 1$, $\mu_i = (\omega_{i_x},\omega_{i_y})$ and $\sigma_i = 0$ for $k$ layers
        \item<2-> instantiate GGX lobes with $F = e_k$, $\omega_{i_k} = reflect(\mu_k)$ and $\alpha_k = f^{-1}(\sigma_k)$
        \item<3-> {
            weighted sum of GGX lobes:\\
            \begin{tabular}{c c c}
                $\rho(\omega_i, \omega_o) = \sum_{k=0}^{N} \frac{e_k G(\omega_{i_k}, \omega_o) D(h_k)}{4 \dotproduct{\omega_{i_k}}{n} \dotproduct{\omega_o}{n}}$
                & &
                \raisebox{-9mm}{\includesvg[width = 0.17\textwidth]{multi_lobe.svg}}
            \end{tabular}
        }
    \end{itemize}
    %%%%%%%%%%%%%%%%%
    \note[item]{Using the adding-doubling method, with \textbf{initial values} of energy as 1, mean as the projected incident direction and a variance of 0, evaluated for all k layers we get a \textbf{vector of statistics}.\click}
    \note[item]{Using them we can \textbf{instantiate k GGX lobes} with the kth energy as the Fresnel term, the reflected kth mean as the incident direction and the inverse mapped variance as the roughness.\click}
    \note[item]{Therefore, the \textbf{resulting BRDF} is a weighted sum of GGX lobes which accurately approximated the appearance of the layered material.}
\end{frame}

%---------------------------------------------------------
\begin{frame}
    \frametitle{Anisotropic interfaces}
    \begin{itemize}
        \setlength\itemsep{1em}
        \item<1-> Approximation using elliptical distribution $\Rightarrow$ two-component vector for variance \parencite{weier2020}
        \item<2-> rotated tangent vector fields for each layer $\Rightarrow$ covariance matrix for variance \parencite{yamaguchi2019}
        \item<3-> precomputation of FGD needs more dimensions or extended split-sum approximation of \textcite{weier2020}
        \item<4-> Monte Carlo integration for image based lighting
    \end{itemize}
    %%%%%%%%%%%%%%%%%
    \note[item]{Using this insight, the variance can be tracked using a two component \textbf{vector} instead of a scalar.\click}
    \note[item]{To also allow \textbf{rotated} tangent vector fields for the individual layers, \textbf{covariance} matrices can be used.\click}
    \note[item]{If \textbf{FGD} should be precomputed, is need more dimensions or an \textbf{extended split sum approximation} that Weier and Belcour introduced can be used, which scales the direction where the textures should be fetched.\click}
    \note[item]{As no \textbf{pre-filtering} of environment maps exists for the anisotropic case, we need to use \textbf{Monte Carlo integration} for image based lighting.}
\end{frame}
