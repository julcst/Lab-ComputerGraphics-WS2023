\subsubsection{Rough reflection}

%---------------------------------------------------------
\begin{frame}
    \frametitle{Reflection by a rough surface}
    \framesubtitle{Energy}
    \textbf{Energy:}\\\bigskip
    Albedo:
    \begin{equation*}
        \FGD = \int_\Omega \rho_{GGX}(\omega_i, \omega_o) \dotproduct{\omega_i}{n} \,d\omega_o
    \end{equation*}
    \pause
    Rendering equation:
    \begin{equation*}
        L_o(\omega_o) = \int_{\Omega} \rho_{GGX}(\omega_i, \omega_o) L_i(\omega_i) \dotproduct{\omega_i}{n} \,d\omega_i 
    \end{equation*}
    \pause
    Decoupling:
    \begin{equation*}
        \int_\Omega L_o(\omega_o) \,d\omega_o \simeq \FGD \times \int_\Omega L_i(\omega_i) \,d\omega_i
    \end{equation*}
    %%%%%%%%%%%%%%%%%
    \note[item]{Notes}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Reflection by a rough surface}
    \framesubtitle{Mean \& Variance}
    \textbf{Mean:} shift towards normal for high roughness, but negligible\\
    \bigskip
    \textbf{Variance:} increases but not linear w.r.t. roughness $\Rightarrow$ linear variance space
    %%%%%%%%%%%%%%%%%
    \note[item]{Notes}
\end{frame}
%---------------------------------------------------------

\begin{frame}
    \frametitle{Reflection by a rough surface}
    \framesubtitle{Summary}
    \begin{equation*}
        \begin{aligned}
            e_r &= e_i \times \FGD\\
            \mu_r &= \reflect(\mu_i)\\
            \sigma_r &= \sigma_i + f(\alpha)
        \end{aligned}
    \end{equation*}
    %%%%%%%%%%%%%%%%%
    \note[item]{Notes}
\end{frame}

\subsubsection{Absorption \& Scattering}

%---------------------------------------------------------
\begin{frame}
    \frametitle{Absorption}
    \framesubtitle{Energy, mean \& variance}
    $\sigma_t = \sigma_a + \sigma_s$\\\bigskip
    \textbf{Energy:} Beer-Lambert's law: 
    \begin{equation*}
        L_o(\omega_o) = \exp{\left(-\sigma_t \frac{h}{\dotproduct{\omega_o}{n}}\right)} L_i(\omega_o)
    \end{equation*} 
    \\\bigskip
    \pause
    \textbf{Mean \& variance:} Impact of absorption negligible
    %%%%%%%%%%%%%%%%%
    \note[item]{Notes}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Scattering}
    \framesubtitle{Energy}
    \textbf{Energy:} amount of radiance scattered accounting for in- and out-scattering: 
    \begin{equation*}
        \begin{aligned}
            L_o(\omega) &= \sigma_s \int_{0}^{h} \exp{\left(-\frac{\sigma_t t}{\dotproduct{\omega}{n}}\right)} L_s(t,\omega) \,dt \\
            \text{with}\\
            L_s(t,\omega) &= \int_{\mathcal{S}} p(\omega_i, \omega) \exp{\left(-\frac{\sigma_t (h-t)}{\dotproduct{\omega_i}{n}}\right)} L_i(\omega_i) \,d\omega_i
        \end{aligned}
    \end{equation*}
    \\\bigskip
    \pause
    Approximation:
    \begin{equation*}
        \int_{\Omega} L_o(\omega) \,d\omega \approx \sigma_s h \exp{\left(-\frac{\sigma_t t}{\dotproduct{\omega}{n}}\right)} \int_{\Omega} L_i(\omega) \,d\omega
    \end{equation*}
    %%%%%%%%%%%%%%%%%
    \note[item]{Notes}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Scattering}
    \framesubtitle{Mean \& variance}
    \textbf{Mean:} Approximation only accounts forward scattering $\Rightarrow$ impact on mean negligible\\\bigskip
    \textbf{Variance:} increases w.r.t. width of phase function\\\bigskip
    Mapping from Henyey-Greenstein phase function anisotropy factor $g$ to variance \parencite{belcour2018}:
    \begin{equation*}
        \sigma_g = \left(\frac{1 - g}{g}\right)^{0.8} \frac{1}{1+g}
    \end{equation*}
    %%%%%%%%%%%%%%%%%
    \note[item]{Notes}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Absorption \& Scattering}
    \framesubtitle{Summary}
    \begin{equation*}
        \begin{aligned}
            e_t &= e_i \times \left[ \left(1 +  \frac{\sigma_s h}{\dotproduct{\omega_o}{n}}\right) \times \exp{\left(-\frac{\sigma_t h}{\dotproduct{\omega_o}{n}}\right)} \right]\\
            \mu_t &= \mu_i\\
            \sigma_t &= \sigma_i + \sigma_g \times \frac{\sigma_s}{\sigma_t}
        \end{aligned}
    \end{equation*}
    %%%%%%%%%%%%%%%%%
    \note[item]{Notes}
\end{frame}

\begin{frame}
    \frametitle{Multiple scattering between two interfaces}
    Account for all $TR^+T$ paths between two interfaces:
    \begin{equation*}
        r_{13} = r_{12} + \sum_{k=0}^{\infty} t_{12} r_{23}^{k+1} r_{21}^{k} t_{21} 
    \end{equation*}
    \\\bigskip
    \pause
    Closed form of this arithmetic series:
    \begin{equation*}
        r_{13} = r_{12} + \frac{t_{12} r_{23} t_{21}}{1 - r_{23} r_{21}}
    \end{equation*}
    \\\bigskip
    \pause
    Similar forms for $r_{31}$, $t_{13}$ and $t_{31}$
    %%%%%%%%%%%%%%%%%
    \note[item]{Notes}
\end{frame}
%---------------------------------------------------------

%---------------------------------------------------------
\begin{frame}
    \frametitle{Adding-doubling for energy}
    \begin{itemize}
        \item<1-> iteratively evaluate $r_{13}$, $r_{31}$, $t_{13}$ and $t_{31}$ \bigskip
        \item<2->{ 
            use statistical operators as coefficients, e.g. for a rough interface:
            \begin{equation*}
                \begin{aligned}
                    r_{12} &= r_{21} = \FGD\\
                    t_{12} &= t_{21} = 1-\FGD
                \end{aligned}
            \end{equation*}
            \bigskip
        }
        \item<3-> doubling: virtual layer twice the depth
    \end{itemize}

\end{frame}

\subsection{BRDF}

%---------------------------------------------------------
\begin{frame}
    \frametitle{BRDF}
    \begin{itemize}
        \setlength\itemsep{1em}
        \item<1-> evaluate adding-doubling method with $e_i = 1$, $\mu_i = (\omega_{i_x},\omega_{i_y})$ and $\sigma_i = 0$ for $k$ layers
        \item<2-> instantiate GGX lobes with $F = e_k$, $\omega_{i_k} = reflect(\mu_k)$ and $\alpha_k = f^{-1}(\sigma_k)$
        \item<3-> {
            weighted sum of GGX lobes:
            \begin{equation*}
                \rho(\omega_i, \omega_o) = \sum_{k=0}^{N} \frac{e_k G(\omega_{i_k}, \omega_o) D(\frac{\omega_{i_k}+\omega_o}{\left\lVert\omega_{i_k}+\omega_o\right\rVert })}{4 \dotproduct{\omega_{i_k}}{n} \dotproduct{\omega_o}{n}}
            \end{equation*}
        }
    \end{itemize}

\end{frame}
